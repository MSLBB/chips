        <input type="number" id="bagsNeeded" required>

        <button type="button" onclick="calculatePotatoes()">Vypočítej počet brambor</button>

    </form>

    <h2 id="result"></h2>


    <script>

        function calculatePotatoes() {

            var chipsPerBag = document.getElementById('chipsPerBag').value;

            var bagsNeeded = document.getElementById('bagsNeeded').value;

            var totalChipsNeeded = chipsPerBag * bagsNeeded;

            var totalPotatoesNeeded = Math.ceil(totalChipsNeeded / 60);


            document.getElementById('result').innerHTML = 'Celkový počet brambor, které potřebujete: ' + totalPotatoesNeeded;

        }

    </script>
